# 农房生成器代码修复指南（对应规则 v3）

本文档列出 `app.js` 中需要修改的具体代码位置及修复方案，以解决卫生间套流、楼梯间尺寸僵化、空间填充不全等问题。

---

## 修复1：楼梯间最小尺寸强制（解决 stairW < 2.4m 漏洞）

### 问题
当前 `layoutFloor1` 中：
```javascript
const stairW = stair ? Math.max(1.5, Math.min(3.5, (stair.targetArea || 10) / frontH)) : 0;
```
允许楼梯间宽度低至 **1.5m**，严重违反 R11/R12（最小净宽 2.4m）。

### 修复方案
将 `resolveCoreLayout` 和 `layoutFloor1` 中的楼梯尺寸计算改为**以最小尺寸为底线，面积自然形成**。

**步骤 A：修改 `resolveCoreLayout`**
```javascript
function resolveCoreLayout(length, width, withStair = true) {
  // v3 修复：楼梯间最小外框 2.4m × 3.6m，禁止硬编码固定面积
  const MIN_STAIR_W = 2.4;
  const MIN_STAIR_H = 3.6;

  let coreW = clamp(length * 0.28, MIN_STAIR_W, Math.min(4.0, length - 2.0));
  let stairH = withStair ? clamp(width * 0.35, MIN_STAIR_H, width * 0.55) : 0;

  // 如果按面积反推导致某一维度过小，优先保证最小尺寸
  if (withStair && stair.targetArea) {
    const desiredW = stair.targetArea / stairH;
    if (desiredW >= MIN_STAIR_W) {
      coreW = clamp(desiredW, MIN_STAIR_W, Math.min(4.0, length - 2.0));
    }
  }

  // 确保楼梯间不超出建筑宽度
  if (withStair && stairH > width - 1.0) {
    stairH = Math.max(MIN_STAIR_H, width - 1.0);
  }

  const leftW = Math.max(2.4, length - coreW);
  const stairY = withStair ? Math.max(0, width - stairH) : width;
  const bathZoneH = Math.max(1, stairY);

  return { coreW, leftW, stairH, stairY, bathZoneH };
}
```

**步骤 B：修改 `layoutFloor1` 中的 stairW 计算**

## 修复2：卫生间独立动线（解决"只能通过厨房进卫生间"）

### 问题
当前 `layoutFloor1` 的方案 B（水平排列后带）中，房间按 `splitBand(..., 'horizontal')` 简单排列：
```
[卧室1] [卧室2] [厨房] [卫生间]
```
如果卫生间在最右侧，它只能通过左侧的厨房进入，严重违反 R6a。

### 修复方案
**在 `layoutFloor1` 的后带布局中，强制卫生间至少与一个非厨房房间（卧室/客厅/过道）相邻。**

**修改 `layoutFloor1` 的后带逻辑：**
```javascript
// ===== 方案B：水平排列（后带高度不足）=====
// 原代码：简单按顺序水平排列

// v3 修复：确保卫生间不处于只能通过厨房到达的位置
const backRooms = [...bedrooms];
if (kitchen) backRooms.push(kitchen);
if (bathroom) backRooms.push(bathroom);
if (storage) backRooms.push(storage);

if (backRooms.length) {
  // 策略：如果卫生间和厨房都存在，强制卫生间排在第一个卧室之后、或最左端
  // 这样卫生间左侧至少有一个卧室邻居
  const ordered = [];
  const bathIdx = backRooms.findIndex(r => r.type === 'bathroom');
  const kitchenIdx = backRooms.findIndex(r => r.type === 'kitchen');

  if (bathIdx >= 0 && kitchenIdx >= 0 && backRooms.length >= 3) {
    // 把卫生间插入到厨房之前，且保证前面至少有一个卧室
    const bedCount = backRooms.filter(r => r.type === 'bedroom').length;
    if (bedCount >= 1) {
      // 顺序：卧室... → 卫生间 → 厨房 → 储藏室
      backRooms.forEach((r, i) => {
        if (r.type === 'bedroom') ordered.push(r);
      });
      ordered.push(backRooms[bathIdx]); // 卫生间
      if (storage) ordered.push(storage);
      ordered.push(backRooms[kitchenIdx]); // 厨房放最后（贴外墙/边缘）
    } else {
      // 无卧室时：卫生间 → 厨房 → 储藏室
      ordered.push(backRooms[bathIdx]);
      ordered.push(backRooms[kitchenIdx]);
      if (storage) ordered.push(storage);
    }
  } else {
    ordered.push(...backRooms);
  }

  result.push(...splitBand(ordered, 0, 0, length, backH, scale, 'horizontal'));
}
```

```javascript
// 在 validateHardConstraints 函数末尾、return violations 之前插入：

// --- R6a: 卫生间独立动线检查 ---
floorPlans.forEach(p => {
  const living = p.rooms.find(r => r.type === 'living_room' || r.type === 'lounge');
  const baths = p.rooms.filter(r => r.type === 'bathroom');
  const kitchens = p.rooms.filter(r => r.type === 'kitchen');
  if (!living || baths.length === 0) return;

  baths.forEach(bath => {
    // BFS 从客厅到卫生间
    const visited = new Set();
    const queue = [living];
    const parent = new Map();
    visited.add(living);
    let found = false;

    while (queue.length > 0 && !found) {
      const cur = queue.shift();
      const neighbors = getRoomNeighbors(cur, p.rooms);
      for (const n of neighbors) {
        if (!visited.has(n)) {
          visited.add(n);
          parent.set(n, cur);
          if (n === bath) { found = true; break; }
          queue.push(n);
        }
      }
    }

    if (!found) {
      addV('R6a', p.floor, bath, '卫生间与客厅不连通，不可达');
      return;
    }

    // 回溯路径，检查是否只经过厨房
    const path = [];
    let node = bath;
    while (node !== living) {
      path.unshift(node);
      node = parent.get(node);
    }
    const passesKitchen = path.some(r => r.type === 'kitchen');
    const nonKitchenNeighbors = path.filter(r => r.type !== 'kitchen');

    // 如果路径中必须经过厨房，且厨房是到达卫生间的唯一入口，则违规
    if (passesKitchen && nonKitchenNeighbors.length === 0) {
      addV('R6a', p.floor, bath, '卫生间只能通过厨房进入，违反独立动线规则');
    }
  });
});
```

## 修复3：楼层空间 100% 填满（自适应面积填充）

### 问题
当前 `allocateRoomAreas` 使用固定 `maxTotalRatio = 0.92`，导致总面积永远 ≤ 92% 可用面积，必然留下 8% 空白。且没有二次平衡机制。

### 修复方案
**步骤 A：修改 `allocateRoomAreas`，增加二次平衡**
```javascript
function allocateRoomAreas(roomList, floorArea) {
  // v3 修复：允许总面积接近 100%，通过 rebalance 二次平衡消除空白
  const maxTotalRatio = 0.98; // 从 0.92 放宽到 0.98（保留 2% 墙体+误差）
  const maxTotalArea = floorArea * maxTotalRatio;

  // ...（保留原有第一步、第二步分配逻辑）...

  const allocated = items.map(item => ({
    ...item,
    targetArea: Math.max(item.hardMin, round2(item.allocated))
  }));

  // v3 新增：二次平衡，填满可用面积
  return rebalanceAreas(allocated, maxTotalArea);
}

// v3 新增函数
function rebalanceAreas(rooms, availableNetArea) {
  const total = rooms.reduce((s, r) => s + (r.targetArea || 0), 0);
  const gap = availableNetArea - total;

  if (Math.abs(gap) < 0.3) return rooms;

  const result = rooms.map(r => ({ ...r }));

  if (gap > 0) {
    // 面积不足：优先放大客厅、主卧、餐厅
    const expandPriority = ['living_room', 'bedroom', 'dining', 'kitchen', 'bathroom', 'study', 'storage'];
    let remaining = gap;
    for (const type of expandPriority) {
      const candidates = result.filter(r => r.type === type && (r.targetArea || 0) < (r.maxA || 999));
      if (!candidates.length) continue;
      const share = remaining / candidates.length;
      candidates.forEach(r => {
        const headroom = (r.maxA || 999) - (r.targetArea || 0);
        const add = Math.min(headroom, share);
        r.targetArea = (r.targetArea || 0) + add;
        remaining -= add;
      });
      if (remaining <= 0.1) break;
    }
  } else {
    // 面积超限：优先压缩储藏室、书房、次卧
    const shrinkPriority = ['storage', 'study', 'bedroom', 'living_room', 'dining'];
    let needShrink = -gap;
    for (const type of shrinkPriority) {
      const candidates = result.filter(r => r.type === type && (r.targetArea || 0) > (r.hardMin || 0));
      if (!candidates.length) continue;
      const share = needShrink / candidates.length;
      candidates.forEach(r => {
        const headroom = (r.targetArea || 0) - (r.hardMin || 0);
        const sub = Math.min(headroom, share);
        r.targetArea = (r.targetArea || 0) - sub;
        needShrink -= sub;
      });
      if (needShrink <= 0.1) break;
    }
  }
  return result;
}
```

## 修复4：上层露台禁止与楼梯间/起居厅重叠（R4b）

### 问题
当前 `layoutUpperFloor` 中，露台和楼梯间/起居厅都放置在 front 带（y 值较大的区域），可能发生投影重叠。

### 修复方案
**修改 `layoutUpperFloor` 的 front 带逻辑，在放置起居厅前检查是否与露台重叠。**

```javascript
// 在 layoutUpperFloor 中，放置起居厅之前：
if (living) {
  let livingX = stairX + stairW;
  let livingW = length - stairW;
  let livingY = stairY;
  let livingH = frontH;

  // v3 修复：若露台在 front 带（tr/tl），必须扣除露台区域，避免重叠
  if (terracePos === 'tr' || terracePos === 'tl') {
    // 如果露台在 front 带，起居厅只能占 front 带减去露台后的剩余部分
    // 且起居厅不可与露台在 x 轴上重叠
    if (terracePos === 'tr') {
      // 露台在右上角：起居厅只能占据 [stairX+stairW, length-terraceW]
      livingW = Math.max(3.5, length - stairW - terraceW);
    } else if (terracePos === 'tl') {
      // 露台在左上角：起居厅从 stairX+stairW 开始，但左侧需留出 terraceW
      // 若 stairX > 0（楼梯不在最左），需额外处理
      livingX = Math.max(stairX + stairW, terraceW);
      livingW = Math.max(3.5, length - livingX);
    }
  }

  // 额外安全检查：若计算出的 livingW 过小，尝试将露台移到 back 带（bl/br）
  if (livingW < 3.5 && terrace) {
    // 强制修改 terracePos 到 back 带
    const backCorners = ['bl', 'br'];
    const newPos = backCorners[Math.floor(Math.random() * backCorners.length)];
    config._terracePos = newPos;
    // 重新计算露台坐标（略，需根据 newPos 重新算 tx, ty, terraceW, terraceDepth）
    // ... 然后重新计算 livingW ...
    livingW = Math.max(3.5, length - stairW);
  }

  if (livingW >= 3.5) {
    pushRoomRectIfValid(result, living, livingX, livingY, livingW, livingH);
  }
}
```

> **注意**：更稳妥的做法是在生成共享露台参数时（`generateBuilding` 中），直接**禁止选择 tr/tl（front 带）**，统一使用 bl/br（back 带），这样从根本上避免重叠。对于农房而言，露台通常是从客厅/卧室向外挑出，放在 back 带（北侧或南侧均可，取决于朝向设定）更合理。

## 修复5：增加过道（走廊）合法化支持（R7a）

### 问题
当前代码没有"过道"这个房间类型。当进深较大时，后带卧室如果垂直排列，它们之间没有交通空间，导致从客厅无法直接进入最内侧的卧室（必须穿越其他卧室）。

## 修复6：门洞生成优化（增加向卧室/过道开卫生间门）

### 问题
当前 `renderPlan` 中的门洞逻辑，`bathroomRooms` 只是简单地找与任意非楼梯/露台房间相邻的边。这可能导致卫生间门开向厨房而不是卧室/过道。

### 修复方案
**修改 `renderPlan` 中卫生间门洞的选择逻辑，优先选择通向客厅/过道/卧室的边。**

```javascript
// 在 renderPlan 中，bathroomRooms 的门洞逻辑：
bathroomRooms.forEach(item => {
  const roomId = item.idx;
  const hasDoor = roomHasPassage(roomId);
  if (hasDoor) return;

  // v3 修复：卫生间门优先顺序：客厅/过道 > 卧室 > 储藏室 > 厨房
  const preferredTypes = ['living_room', 'lounge', 'corridor', 'bedroom', 'storage', 'kitchen'];
  let picked = null;
  for (const pt of preferredTypes) {
    const options = (edgeOfRoom.get(roomId) || []).filter(edge => {
      const other = plan.rooms[getOtherId(edge, roomId)];
      return other && other.type === pt;
    });
    if (options.length) {
      picked = pickBest(options, edge => edge.overlap);
      if (picked) break;
    }
  }

  if (picked) {
    addDoorFromEdge(picked, roomId);
  } else {
    // fallback：任意可用邻居
    const fallback = (edgeOfRoom.get(roomId) || []).filter(edge => {
      const other = plan.rooms[getOtherId(edge, roomId)];
      return other && other.type !== 'stairs' && other.type !== 'terrace';
    });
    const fb = pickBest(fallback, edge => edge.overlap);
    if (fb) addDoorFromEdge(fb, roomId);
  }
});
```
